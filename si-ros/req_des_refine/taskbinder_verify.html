theory taskbinder_verify

imports 
      "scheduler_verify" 
      "../des/des_taskbinder"

begin

(* 
  Refinement Verification of Function pushSwarmAcbs
*)
lemma req_des_pushSwarmAcbs_aux0 : "
                NonDetMonad_call.call  (Des_appendSwarmACBs sw al)
                  (λσ. (pActorScheduler (fst σ), snd σ)) 
                  (λσ s. ((fst σ)⦇pActorScheduler := fst s⦈, snd s) )  =
                call_spec (Des_appendSwarmACBs sw al)
                  (λσ. (pActorScheduler (fst σ), snd σ)) 
                  (λσ s. ((fst σ)⦇pActorScheduler := fst s⦈, snd s) )
                  (λs. fst s = pActorScheduler tb)
                  (λa s. fst s = Re_appendSwarmACBs (pActorScheduler tb) sw al ∧
                         a = () )"
  by (simp add: call_spec_def)
lemma req_des_pushSwarmAcbs :
       "
       ⦃λs. (fst s) = tb⦄ 
       Des_pushSwarmAcbs sw al
       ⦃λr s. fst s = fst (Re_pushSwarmAcbs tb sw al) ∧
              r =  snd (Re_pushSwarmAcbs tb sw al)⦄"
  unfolding Des_pushSwarmAcbs_def Re_pushSwarmAcbs_def case_prod_unfold
  apply(subst req_des_pushSwarmAcbs_aux0[where tb = "tb"])
  apply wpsimp
  apply (insert req_des_appendSwarmACBs)
  apply simp
  by simp
  

(* 
  Refinement Verification of Function bindCallback
*)
lemma req_des_bindCallback_aux0 : 
  " ¬ snd (Re_pushSwarmAcbs tb (fst req) (snd req)) ⟶ ⦃λs. fst s = tb⦄ Des_pushSwarmAcbs (fst req) (snd req) 
    ⦃λr s. ¬ r ∧ (¬ r ⟶ fst s = tb)⦄
  "
  using req_des_pushSwarmAcbs
  unfolding valid_def
  by (simp add: Re_pushSwarmAcbs_def case_prod_beta')
lemma req_des_bindCallback_aux1 : 
  " snd (Re_pushSwarmAcbs tb (fst req) (snd req)) ⟶ ⦃λs. fst s = tb⦄ Des_pushSwarmAcbs (fst req) (snd req) 
    ⦃λr s. (r ⟶ fst s = fst (Re_pushSwarmAcbs tb (fst req) (snd req))) ∧ r⦄
  "
  using req_des_pushSwarmAcbs
  unfolding valid_def
  by (simp add: Re_pushSwarmAcbs_def case_prod_beta')
lemma req_des_bindCallback :
       "
       ⦃λs. (fst s) = tb⦄ 
       Des_bindCallback req
       ⦃λr s. fst s = fst (Re_bindCallback tb req) ∧
              r =  snd (Re_bindCallback tb req)⦄"
  unfolding Des_bindCallback_def Re_bindCallback_def Let_def case_prod_unfold
  apply wpsimp apply rule
  apply (wpsimp wp: req_des_bindCallback_aux0)
  apply (wpsimp wp: req_des_bindCallback_aux1)
  by simp


(* 
  Refinement Verification of Function startService
*)
lemma serviceEqual_bindCallback : "serviceEqual Re_bindCallback Des_bindCallback"
  using req_des_bindCallback
  unfolding serviceEqual_def valid_def
  by simp
lemma req_des_startService_aux0 : 
  "   ⦃λs. fst s = fst tR ∧ rosEqual (snd tR) (snd s) ∧ nha = (nh (fst tR)) ⦄ 
      Des_advertiseService nha ''task_bind_service'' Des_bindCallback 
      ⦃λss s.
           fst s⦇task_bind_server := ss⦈ = fst tR⦇task_bind_server :=
                   snd (Re_advertiseService (nh (fst tR)) (snd tR) ''task_bind_service'' Re_bindCallback)⦈ 
           ∧
           rosEqual
           (fst (Re_advertiseService (nh (fst tR)) (snd tR) ''task_bind_service''
                Re_bindCallback))
           (snd s⦇logFile := Ros_INFO (logFile (snd s)) ''[Task Bind] Service initialized.''⦈)⦄
  "
  apply (rule req_des_appendSwarmACBs_aux1)
  subgoal
    unfolding Des_advertiseService_def modify_def
    apply wpsimp
    by (simp add:  rosEqual_def Des_advertiseService_def Re_advertiseService_def Let_def Re_getPath_def  Des_getPath_def)
  using req_des_advertiseService serviceEqual_bindCallback
  unfolding valid_def rosEqual_def
  by fastforce
lemma req_des_startService :
       "
       ⦃λs. (fst s) = (fst tR) ∧ rosEqual (snd tR) (snd s)⦄ 
       Des_startService
       ⦃λr s. fst s = fst (Re_startService tR) ∧
              rosEqual (snd (Re_startService tR)) (snd s)
       ⦄"
  unfolding Des_startService_def Re_startService_def Let_def case_prod_unfold
  apply wpsimp
  apply (wpsimp wp: req_des_startService_aux0)
  apply wpsimp
  by simp


end
