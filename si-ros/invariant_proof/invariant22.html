lemma Re_api_terminate_inv23 : "invariant_23 sche ⟹ invariant_23 (Re_api_terminate sche bName actID pName)"
  unfolding  Re_api_terminate_def 
  apply (subgoal_tac "invariant_23 sche ⟹ invariant_23 (fst ( api_updateSchedule sche bName actID pName pluginState.ACTOR_STATE_FINISH))")
  apply (smt Re_api_appendScheduleRequest_inv23 case_prod_unfold)
  apply (simp add:  api_updateSchedule_def invariant_23_def Let_def case_prod_unfold)
  proof -
  assume a1: "∀bus∈set (registeredBus sche). ∀plu∈set (pluginList bus). plugin.actorID plu ≤ actorNo sche"
  { fix pp :: plugin and bb :: bus
    have "∀bs b. ∃n. ((b::bus) ∉ set bs ∨ n < length bs) ∧ (b ∉ set bs ∨ bs ! n = b)"
      by (metis (lifting) in_set_conv_nth)
    then obtain nn :: "bus list ⇒ bus ⇒ nat" where
      ff1: "⋀b bs. (b ∉ set bs ∨ nn bs b < length bs) ∧ (b ∉ set bs ∨ bs ! nn bs b = b)"
      by metis
    then have ff2: "⋀bs n b ba. nn (bs[n := b]) ba ≠ n ∨ ¬ n < length bs ∨ ba ∉ set (bs[n := b]) ∨ b = ba"
  by (metis (lifting) nth_list_update)
    have ff3: "⋀n bs b ba. ¬ n < length bs ∨ b ∉ set (bs[n := ba]) ∨ nn (bs[n := ba]) b = n ∨ b ∈ set bs"
      using ff1 by (metis (no_types) length_list_update nth_list_update nth_mem)
    { assume "find_index (λb. bus.name b = bName) (registeredBus sche) < length (registeredBus sche)"
  have "fst (api_updateBus (registeredBus sche ! find_index (λb. bus.name b = bName) (registeredBus sche)) actID pName pluginState.ACTOR_STATE_FINISH) = bb ⟶ pp ∉ set (pluginList bb) ∨ plugin.actorID pp ≤ actorNo sche ∨ ¬ find_index (λb. bus.name b = bName) (registeredBus sche) < length (registeredBus sche) ∨ bb ∉ set ((registeredBus sche) [find_index (λb. bus.name b = bName) (registeredBus sche) := fst (api_updateBus (registeredBus sche ! find_index (λb. bus.name b = bName) (registeredBus sche)) actID pName pluginState.ACTOR_STATE_FINISH)])"
    using a1 by (meson Re_apiUpdateBus_aux nth_mem)
    then have "pp ∉ set (pluginList bb) ∨ plugin.actorID pp ≤ actorNo sche ∨ ¬ find_index (λb. bus.name b = bName) (registeredBus sche) < length (registeredBus sche) ∨ bb ∉ set ((registeredBus sche) [find_index (λb. bus.name b = bName) (registeredBus sche) := fst (api_updateBus (registeredBus sche ! find_index (λb. bus.name b = bName) (registeredBus sche)) actID pName pluginState.ACTOR_STATE_FINISH)])"
  using ff3 ff2 a1 by meson }
    then have "pp ∉ set (pluginList bb) ∨ plugin.actorID pp ≤ actorNo sche ∨ ¬ find_index (λb. bus.name b = bName) (registeredBus sche) < length (registeredBus sche) ∨ bb ∉ set ((registeredBus sche) [find_index (λb. bus.name b = bName) (registeredBus sche) := fst (api_updateBus (registeredBus sche ! find_index (λb. bus.name b = bName) (registeredBus sche)) actID pName pluginState.ACTOR_STATE_FINISH)])"
      by (metis (lifting)) }
  then show "find_index (λb. bus.name b = bName) (registeredBus sche) < length (registeredBus sche) ⟶ (∀b∈set ((registeredBus sche) [find_index (λb. bus.name b = bName) (registeredBus sche) := fst (api_updateBus (registeredBus sche ! find_index (λb. bus.name b = bName) (registeredBus sche)) actID pName pluginState.ACTOR_STATE_FINISH)]). ∀p∈set (pluginList b). plugin.actorID p ≤ actorNo sche)"
    by metis
  qed
