  proof -

    { fix pp :: plugin and bb :: bus
      { assume "find_index (λa. Re_switchAcb1 a anme) (acbList sche) < length (acbList sche) ∧ ¬ Re_switchAcb2 (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche)) aid"
        moreover
        { assume "plugin.actorID pp ∈ ACB.actorID ` set (acbList sche) ∧ find_index (λa. Re_switchAcb2 a aid) (acbList sche) ≠ find_index (λa. Re_switchAcb1 a anme) (acbList sche)"
          then have "plugin.state pp = pluginState.ACTOR_STATE_FINISH ∨ pp ∉ set (pluginList bb) ∨ bb ∉ set (registeredBus sche) ∨ ¬ find_index (λa. Re_switchAcb1 a anme) (acbList sche) < length (acbList sche) ∨ ACB.state (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche)) = actorState.ACTOR_STATE_RUNNING ∨ plugin.actorID pp ∈ ACB.actorID ` set ((acbList sche) [find_index (λa. Re_switchAcb1 a anme) (acbList sche) := (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche)) ⦇prio := prio (acbList sche ! find_index (λa. Re_switchAcb2 a aid) (acbList sche))⦈, find_index (λa. Re_switchAcb2 a aid) (acbList sche) := (acbList sche ! find_index (λa. Re_switchAcb2 a aid) (acbList sche)) ⦇prio := prio (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche))⦈])"
            by (simp add: image_list_update) }
        ultimately have "plugin.state pp = pluginState.ACTOR_STATE_FINISH ∨ pp ∉ set (pluginList bb) ∨ bb ∉ set (registeredBus sche) ∨ ¬ find_index (λa. Re_switchAcb1 a anme) (acbList sche) < length (acbList sche) ∨ ACB.state (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche)) = actorState.ACTOR_STATE_RUNNING ∨ plugin.actorID pp ∈ ACB.actorID ` set ((acbList sche) [find_index (λa. Re_switchAcb1 a anme) (acbList sche) := (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche)) ⦇prio := prio (acbList sche ! find_index (λa. Re_switchAcb2 a aid) (acbList sche))⦈, find_index (λa. Re_switchAcb2 a aid) (acbList sche) := (acbList sche ! find_index (λa. Re_switchAcb2 a aid) (acbList sche)) ⦇prio := prio (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche))⦈])"
          using a1 by (metis nth_find_index) }
      then have "plugin.state pp = pluginState.ACTOR_STATE_FINISH ∨ pp ∉ set (pluginList bb) ∨ bb ∉ set (registeredBus sche) ∨ ¬ find_index (λa. Re_switchAcb1 a anme) (acbList sche) < length (acbList sche) ∨ ACB.state (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche)) = actorState.ACTOR_STATE_RUNNING ∨ plugin.actorID pp ∈ ACB.actorID ` set ((acbList sche) [find_index (λa. Re_switchAcb1 a anme) (acbList sche) := (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche)) ⦇prio := prio (acbList sche ! find_index (λa. Re_switchAcb2 a aid) (acbList sche))⦈, find_index (λa. Re_switchAcb2 a aid) (acbList sche) := (acbList sche ! find_index (λa. Re_switchAcb2 a aid) (acbList sche)) ⦇prio := prio (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche))⦈])"
        using Re_switchAcb2_def by blast }
    then show "find_index (λa. Re_switchAcb2 a aid) (acbList sche) < length (acbList sche) ⟶ find_index (λa. Re_switchAcb1 a anme) (acbList sche) < length (acbList sche) ∧ ACB.state (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche)) ≠ actorState.ACTOR_STATE_RUNNING ⟶ (∀b∈set (registeredBus sche). ∀p∈set (pluginList b). plugin.state p ≠ pluginState.ACTOR_STATE_FINISH ⟶ plugin.actorID p ∈ ACB.actorID ` set ((acbList sche) [find_index (λa. Re_switchAcb1 a anme) (acbList sche) := (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche)) ⦇prio := prio (acbList sche ! find_index (λa. Re_switchAcb2 a aid) (acbList sche))⦈, find_index (λa. Re_switchAcb2 a aid) (acbList sche) := (acbList sche ! find_index (λa. Re_switchAcb2 a aid) (acbList sche)) ⦇prio := prio (acbList sche ! find_index (λa. Re_switchAcb1 a anme) (acbList sche))⦈]))"
      by blast
  qed
