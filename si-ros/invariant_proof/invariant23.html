lemma Re_activateActor_inv24 : "invariant_24 sche ⟹ invariant_24 (fst (Re_activateActor sche aid anme))"
  unfolding  Let_def Re_activateActor_def Re_activateAcb1_def Re_activateAcb2_def 
  apply auto
  apply (subgoal_tac "∀ i<length (acbList sche). ∀ j < length (acbList sche).
          invariant_24 (sche ⦇acbList := (acbList sche)
              [i:= (acbList sche!i)⦇prio := prio (acbList sche!j)⦈]⦈)")
   apply blast
  apply (simp add:invariant_24_def)
proof -
assume a1: "∀acb∈set (acbList sche). ACB.state acb = actorState.ACTOR_STATE_INIT ⟶ (∀bus∈set (registeredBus sche). ACB.actorID acb ∉ plugin.actorID ` set (pluginList bus))"
{ fix aa :: ACB and bb :: bus and nn :: nat and nna :: nat
{ assume "aa ∉ set (acbList sche)"
  have "(ACB.state aa ≠ actorState.ACTOR_STATE_INIT ∨ bb ∉ set (registeredBus sche) ∨ ¬ nn < length (acbList sche) ∨ ACB.actorID aa ∉ plugin.actorID ` set (pluginList bb) ∨ aa ∉ set ((acbList sche) [nn := (acbList sche ! nn) ⦇prio := prio (acbList sche ! nna)⦈])) ∨ aa ∈ set (acbList sche) ∨ (∃a. acbList sche ! nn = a ∧ a ∉ set (acbList sche))"
    using a1 by (metis (no_types) Re_ActorPrioChange_inv13_aux Re_ActorPrioChange_inv6_aux index_less_size_conv index_update_if_diff2 length_list_update)
then have "(ACB.state aa ≠ actorState.ACTOR_STATE_INIT ∨ bb ∉ set (registeredBus sche) ∨ ¬ nn < length (acbList sche) ∨ ACB.actorID aa ∉ plugin.actorID ` set (pluginList bb) ∨ aa ∉ set ((acbList sche) [nn := (acbList sche ! nn) ⦇prio := prio (acbList sche ! nna)⦈])) ∨ aa ∈ set (acbList sche)"
  by (meson in_set_conv_nth) }
    then have "ACB.state aa ≠ actorState.ACTOR_STATE_INIT ∨ bb ∉ set (registeredBus sche) ∨ ¬ nn < length (acbList sche) ∨ ACB.actorID aa ∉ plugin.actorID ` set (pluginList bb) ∨ aa ∉ set ((acbList sche) [nn := (acbList sche ! nn) ⦇prio := prio (acbList sche ! nna)⦈])"
  using a1 by blast }
then show "∀n<length (acbList sche). ∀na<length (acbList sche). ∀a∈set ((acbList sche) [n := (acbList sche ! n) ⦇prio := prio (acbList sche ! na)⦈]). ACB.state a = actorState.ACTOR_STATE_INIT ⟶ (∀b∈set (registeredBus sche). ACB.actorID a ∉ plugin.actorID ` set (pluginList b))"
  by blast
qed
