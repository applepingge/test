(* 
  Design Definition of Function createInstance

definition Des_createInstance :: "string ⇒ ((bus × RosLog), string + bool)nondet_monad"
  where "Des_createInstance plugName ≡ 
    let plug = nullPlugin⦇plugin.name := plugName⦈ 
    in if plugName = []
       then throwError ''emptyNameError''
       else 
        do 
           modify (λσ. ((fst σ)⦇pluginList := (pluginList (fst σ)) @ [plug]⦈, snd σ) );
           returnOk True
        od "


(* 
  Design Definition of Function loadPlugin
*)
(* st ← return (''[General Bus]Failed to create the '' @ pName @ '' device, are you sure it is properly registered and that the containing library is built? Exception: '' @ e);*)
definition Des_loadPlugin :: "string ⇒ int64_t ⇒ ((bus × RosLog), bool) nondet_monad"
  where "Des_loadPlugin pName actID ≡ 
    do 
      pluginNum ← gets (λσ. nRegisteredPlugin (fst σ));
      if pluginNum < PLUGIN_NUM_MAX-1
      then 
        do res ←(do 
                tmp ← Des_createInstance pName;
                return tmp
              od) <catch> (λe. do
                                  st ← return (''[General Bus]Failed to create the '' @ pName @ '' device,  Exception: '' @ e);
                                  modify (λσ. (fst σ, Ros_FATAL (snd σ) st));
                                  return False
                               od);
        if ¬res
        then return False
        else 
          do
             modify (λσ. ( (fst σ)⦇pluginOwner := (pluginOwner (fst σ)) @ [actID]⦈, snd σ) );
             modify (λσ. ( (fst σ)⦇pluginName := (pluginName (fst σ)) @ [pName]⦈, snd σ) );
             st ← gets (λσ. ''[General Bus]Actor '' @ (show actID) @ '' is loading plugin  '' @ pName @ '' on '' @ (bus.name (fst σ)) );
             modify (λσ. (fst σ, Ros_INFO (snd σ) st));
             modify (λσ. let pList = pluginList (fst σ);
                             num = nat (nRegisteredPlugin (fst σ))
                         in
                             ((fst σ)⦇pluginList := pList[num := pluInitialize (pList!num) actID pName]⦈, snd σ)
             ); 
             modify (λσ. ((fst σ)⦇nRegisteredPlugin := nRegisteredPlugin (fst σ) +1⦈, snd σ) );
             return True
          od
        od
      else return False
    od"


(* 
  Design Definition o
f Function startPlugin
*)
definition Des_startPlugin :: "string ⇒ int64_t ⇒ ((bus × RosLog),bool) nondet_monad"
  where "Des_startPlugin pName actID  ≡ 
    do 
      i ← return 0;
      i ← (whileLoopE (λi σ. int i < nRegisteredPlugin (fst σ))
            (λi. do 
               p ← gets (λσ. (pluginList (fst σ))!i);
               if getPlugName p = pName ∧ getPlugActorID p = actID
               then 
                  do                     
                    retp ← NonDetMonad_call.call Des_resume (λσ. (p,snd σ))
                       (λσ p. ((fst σ)⦇pluginList := list_update (pluginList (fst σ)) i (fst p)⦈, snd p) );
                    throwError i 
                  od
               else returnOk (i+1)
            od) 
            (i) 
           ) <catch> (λe. return e);
      condition (λσ. int i = nRegisteredPlugin (fst σ)) 
        (do
            st ← return (''[General Bus]Plugin %s not loaded'' @ pName );
            modify (λσ. (fst σ, Ros_WARN (snd σ) st));
            return False
         od) 
        (do 
            st ← gets (λσ. ''[General Bus]Actor '' @ show actID @ '' is starting plugin '' @ pName @ '' on '' @ (bus.name (fst σ)));
            modify (λσ. (fst σ, Ros_INFO (snd σ) st));
            return True
         od)
    od"

(* 
  Design Definition of Function loadPluginInfoList
*)
definition Des_loadPluginInfoList :: "PluginInfo list ⇒ int64_t ⇒ ((bus ×  RosLog), bool) nondet_monad"
  where "Des_loadPluginInfoList pil actID ≡
    do 
      setResult ← return True;
      res ← (whileLoopE (λ(i,setResult) σ. i < length pil)
              (λ(i,setResult). do 
               tBusName ← return (busName (pil!i));
               setResult ← condition (λσ. bus.name (fst σ) = tBusName)
                  (do 
                    tmpRes ← Des_loadPlugin (PluginInfo.name (pil!i)) actID;
                    return (setResult ∧ tmpRes)
                   od) (return setResult);
               if setResult = False
               then throwError i
               else returnOk (i+1,setResult)
              od) (0,True)) <catch> (λe. return (e,False));
       return (snd res)
    od"
definition Des_updateActorPluginState :: "ActorPluginInfo ⇒ actorState ⇒ ((bus × RosLog), bool) nondet_monad"
  where "Des_updateActorPluginState api st ≡ 
    do
      setResult ← return True;
      setResult ← (case st of 
          actorState.ACTOR_STATE_INIT ⇒ 
             (do
               Result ← Des_loadPluginInfoList (plugins (ActorPluginInfo.taskInfo api)) (ActorPluginInfo.actorID api);
               return Result
             od) |
          actorState.ACTOR_STATE_RUNNING ⇒ 
             (do 
               setResult ← if ActorPluginInfo.state api = actorState.ACTOR_STATE_INIT
                            then
                             do 
                                setResult ← Des_loadPluginInfoList (plugins (ActorPluginInfo.taskInfo api)) (ActorPluginInfo.actorID api);
                                return setResult
                             od
                            else return setResult;
               condition (λs. setResult) 
                  (Des_startActorPlugins (ActorPluginInfo.actorID api))
                  (return False)
              od) |
          actorState.ACTOR_STATE_PAUSE ⇒
              (do 
               setResult ← Des_pauseActorPlugins (ActorPluginInfo.actorID api);
               return setResult
              od) |
          actorState.ACTOR_STATE_FINISH ⇒
              (do 
               setResult ← Des_stopActorPlugins (ActorPluginInfo.actorID api);
               return setResult
              od) |
          _ ⇒ return False 
      );
      return setResult
    od"

(* 
  Design Definition of Function updateActorPluginStates
*)
definition Des_updateActorPluginStates :: "ActorPluginInfo list ⇒ actorState ⇒ ((bus × RosLog), bool) nondet_monad"
  where "Des_updateActorPluginStates apil st ≡ 
    do
      setResult ← return True; 
      res ← (whileLoopE (λ(i,setResult) σ. i < length apil)
              (λ(i,setResult). 
                 if st = ActorPluginInfo.state (apil!i) 
                 then returnOk (i+1,setResult)
                 else 
                  (do 
                    tmpRes ← Des_updateActorPluginState (apil!i) st;
                    setResult1 ← return (setResult ∧ tmpRes);
                    if ¬setResult1 
                    then throwError i
                    else returnOk (i+1,setResult1)
                   od)
              ) 
              (0,setResult) 
             ) <catch> (λe. return (e,False)) ; 
       return (snd res)
    od"


(* 
  Design Definition of Function getActorPlugin
*)
definition Des_getActorPlugin :: " int64_t ⇒ string ⇒ ((bus × RosLog), (plugin × bool)) nondet_monad"
  where "Des_getActorPlugin actID plugName ≡
    do
       res ← (whileLoopE (λ(i,plu) σ. (int)i < nRegisteredPlugin (fst σ))
              (λ(i,plu). 
                 condition (λσ. ((pluginOwner (fst σ)) ! i) = actID ∧ ((pluginName (fst σ)) ! i) = plugName)
                 (do
                  plu ← gets(λσ. (pluginList (fst σ)) ! i);
                  throwError (i, plu)
                  od)
                 (returnOk (i+1, plu))
              ) 
              (0, nullPlugin) 
       ) <catch> (λe. return e);
       condition (λσ. nRegisteredPlugin (fst σ) = int (fst res)) 
        (return ((snd res), False)) 
        (return ((snd res), True))
    od"


(* 
  Design Definition of Function getActorPluginState
*)
definition Des_getActorPluginState :: "string ⇒ int64_t ⇒ ((bus × RosLog), pluginState) nondet_monad"
  where "Des_getActorPluginState plugName actID ≡
    do
       name ← return plugName;
       res ← (whileLoopE (λ(i,st) σ. (int)i < nRegisteredPlugin (fst σ))
              (λ(i,st). 
                 condition (λσ. ((pluginOwner (fst σ)) ! i) = actID ∧ ((pluginName (fst σ)) ! i) = name)
                 (do
                  st ← gets(λσ. getPlugState ((pluginList (fst σ)) ! i));
                  throwError (i, st)
                  od)
                 (returnOk (i+1, st))
              ) 
              (0, pluginState.NOT_VALID) 
       ) <catch> (λe. return e);
       return (snd res)
    od"
