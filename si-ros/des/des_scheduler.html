(* 
  Design Definition of Function yieldActor
*)
definition Des_yieldActor :: "int64_t ⇒ ((scheduler × RosLog), bool)nondet_monad"
  where "Des_yieldActor anActorID ≡ 
   do
     len ← gets (λσ. length (acbList (fst σ)));
     res ← (whileLoopE (λi σ. i ≥ 0 ∧ i < len) 
              (λi. do
                 p ← gets (λσ. (acbList (fst σ))!i);
                 if ACB.actorID p = anActorID ∧ ACB.state p = actorState.ACTOR_STATE_RUNNING
                 then(     
                      if  ACB.prio p ≥ MAX_PRIORITY
                      then 
                        do
                          st ← gets (λσ. ''[Actor Scheduler] The actor '' @ show (ACB.prio ((acbList (fst σ)) ! i))  @ '' prio >= MAX_PRIORITY'');
                          modify (λσ. (fst σ, Ros_ERROR (snd σ) st));
                          throwError (0)
                        od
                      else 
                        (do
                            modify (λσ. ((fst σ)⦇acbList := list_update (acbList (fst σ)) i 
                                      (p⦇prio := prio p + 1⦈) ⦈, snd σ) ); 
                            throwError (len+1)                     
                         od)
                    )
                else returnOk (i+1)           
              od) 
              (0) 
            ) <catch> (λe. return e);
     condition (λσ.  res = len+1)
       (return True)
       (do
            st ← return (''[Actor Scheduler] Cannot find the running actor by ID '' @ show anActorID);
            modify (λσ. (fst σ, Ros_ERROR (snd σ) st));
            return False
       od)
   od"

               pi ← gets (λσ. acbList (fst σ)! (nat i));

                      resj ← (whileLoopE (λj σ. j≥0 ∧ j< int len)

                                 pj ← gets (λσ. (acbList (fst σ))!(nat j));
                                 if ACB.name pj = anActorName ∧ swarmID (ACB.pSwarmInfo pj) = swarmID (ACB.pSwarmInfo pi)

                                    if ACB.state pj = actorState.ACTOR_STATE_RUNNING

                                      do 
                                        st ← return (''[Actor Scheduler] Try to activate actor which is already running'');

                                     (do
                                       modify (λσ. ((fst σ)⦇acbList := list_update (acbList (fst σ)) (nat j) (pj⦇prio := prio pi⦈) ⦈, snd σ) ); 
                                       throwError (-1)
                                     od)
                                 else returnOk (j+1)
                               od) 
                               (0::int) 
                                ) <catch> (λe. return e);
                        condition(λσ. resj = -1)
                          (throwError (-1))

                            st ← return (''[Actor Scheduler] Cannot find the actor by name '' @ anActorName);
                            modify (λσ. (fst σ, Ros_ERROR (snd σ) st));
                            throwError (-2)

(* 
  Design Definition of Function getActorPlugin_sche
*)
definition Des_getActorPlugin_sche :: "int64_t ⇒ string ⇒ ((scheduler × RosLog), (plugin × bool))nondet_monad"
  where "Des_getActorPlugin_sche anActorID aPluginName ≡ 
  do
    leni ← gets (λσ. length (acbList (fst σ)));
    resi ← (whileLoopE (λ(i,plu,flag) σ. i < leni)
      (λ(i,plu,flag). do
         ai ← gets (λσ. (acbList (fst σ))! i);
         if ACB.actorID ai = anActorID
         then 
           (do
              ptrTaskInfo ← return (ACB.taskInfo ai);
              lenj ← return (length (plugins ptrTaskInfo));
              resj ←(whileLoopE (λ(j,plu,flag) σ. j < lenj)
                (λ(j,plu,flag). do
                   pj ← return ((plugins ptrTaskInfo)! j);
                   busName ← return (PluginInfo.busName pj);
                   pluginName ← return (PluginInfo.name pj);
                   if pluginName = aPluginName 
                   then 
                     (do 
                        lenk ← gets (λσ. length (registeredBus (fst σ)));
                        resk ←(whileLoopE (λ(k,plu,flag) σ. k < lenk)
                           (λ(k,plu,flag). do
                              bk ← gets (λσ. (registeredBus (fst σ))! k);
                              tempStr ← return (getBusName bk);
                              if busName = tempStr
                              then 
                                (do
                                   ret ← NonDetMonad_call.call (Des_getActorPlugin anActorID aPluginName) 
                                          (λσ. (bk,snd σ)) (λσ b. σ);
                                    throwError (lenk ,ret)
                                 od)
                              else returnOk(k+1, plu, flag)
                           od) 
                           (0,nullPlugin,False) 
                           ) <catch> (λe. return e);
                        throwError (lenj, snd resk)
                      od)
                   else returnOk (j+1,plu,flag)
                 od) 
                 (0,nullPlugin,False)
                 ) <catch> (λe. return e);
                throwError (leni, snd resj)
            od)
         else returnOk (i+1,plu,flag)
       od) 
       (0,nullPlugin,False)) <catch> (λe. return e);
    return (snd resi)
  od"

(* 
  Design Definition of Function appendACBs
*)
definition Des_appendACBs :: "ACB list ⇒ SwarmInfo ⇒ ((scheduler × RosLog),int)nondet_monad" 
  where "Des_appendACBs anACBs anPSwarm ≡
    do
     rest ←(whileLoopE (λ(t,tb) σ. t < length anACBs)
             (λ(t,tb). 
               do
                unknownBus ← return '''';
                acb ← return (anACBs!t);
                resi ←(whileLoopE (λ(i,ib) σ.  i < length (TaskInfo.plugins (ACB.taskInfo acb)))
                        (λ(i,ib). do
                          resj ← (whileLoopE (λj σ.  j <  length (registeredBus (fst σ)))
                                   (λj. do
                                      tempStr ← gets (λσ. getBusName ((registeredBus (fst σ))!j));
                                      p ← return ((TaskInfo.plugins (ACB.taskInfo acb))!i);                                      
                                      unknownBus ← return (PluginInfo.busName p);
                                      if tempStr = unknownBus 
                                      then throwError j
                                      else returnOk (j+1)
                                    od) (0)) <catch> (λe. return e);
                          condition (λσ. length (registeredBus (fst σ)) = resj)
                           (do
                              st ← return ( ''[Actor Scheduler] appendACBs: unknown bus '' @ unknownBus @ '', deleting actor '' @ ACB.name (anACBs ! t) @ ''...'');
                              modify (λσ. (fst σ, Ros_ERROR (snd σ) st)); 
                              throwError (i,False)
                            od)
                           (returnOk (i+1,True))
                         od) (0,True) ) <catch> (λe. return e);
                if snd resi
                then returnOk (t+1,True)
                else throwError (t,False)
              od) 
             (0,True) 
             ) <catch> (λe. return e);
    if snd rest = False
    then return 1
    else  
       (do
       whileLoop (λt σ.  t < length anACBs)
        (λt. do
          modify (λσ. ((fst σ)⦇actorNo := actorNo (fst σ) +1⦈, snd σ));
          acb ← return (anACBs!t);
          acb ← gets (λσ. acb⦇ACB.actorID := actorNo (fst σ)⦈);
          acb ← return (acb⦇ACB.state := actorState.ACTOR_STATE_INIT⦈ );
          acb ← return (acb⦇pSwarmInfo := anPSwarm⦈ );
          modify (λσ. ((fst σ)⦇acbList := acbList (fst σ) @ [acb]⦈, snd σ) );
          return (t+1)
        od) 
        (0);
       return 0
      od)
    od"


(* 
  Design Definition of Function appendSwarmACBs
*)
definition Des_appendSwarmACBs :: " SwarmInfo ⇒ ACB list ⇒ ((scheduler × RosLog),unit)nondet_monad"
  where "Des_appendSwarmACBs anSwarm anACBs ≡
   do
     swarmNum ← gets (λσ. nSwarm (fst σ));
     if swarmNum  + 1 ≤ SWARM_MAX_SIZE
     then
       do
         modify (λσ. ((fst σ)⦇nSwarm := nSwarm (fst σ) +1⦈, snd σ) );
         anSwarm ← gets (λσ. anSwarm⦇swarmID := nSwarm (fst σ)⦈);
         anSwarm ← return (anSwarm⦇SwarmInfo.state := actorState.ACTOR_STATE_INIT⦈);
         modify (λσ. ((fst σ)⦇swarmList := swarmList (fst σ) @ [anSwarm] ⦈, snd σ) );
         tmp ← Des_appendACBs anACBs anSwarm;
         if tmp = 0
         then
           (do
              appendMsg ← return nullScheduleMsg;
              appendMsg ← return (appendMsg⦇type := SCHEDULE_CMD_RAW⦈);
              appendMsg ← return (appendMsg⦇param1 := 0⦈);
              appendMsg ← return (appendMsg⦇param2 := 0⦈);
              Des_appendScheduleMsg appendMsg
            od) 
         else return ()
       od
     else return ()
   od"


(* 
  Design Definition of Function anSwarmName
*)
definition Des_getSwarm :: "string ⇒ ((scheduler × RosLog), SwarmInfo×bool)nondet_monad"
  where "Des_getSwarm anSwarmName ≡
    do
      anSwarm ← return nullSwarmInfo;
      res ← (whileLoopE (λi σ. int (fst i) < nSwarm (fst σ))
              (λi. 
                condition (λσ. anSwarmName=swarmName (swarmList (fst σ)!(fst i)))
                (do
                  anSwarm ← gets (λσ. swarmList (fst σ)!(fst i));
                  throwError (fst i,anSwarm)
                 od)
                (returnOk ((fst i)+1,anSwarm))
               ) (0,anSwarm))
             <catch> (λe. return (e));
      condition(λσ. fst res ≠ nat (nSwarm (fst σ)))
       (return (snd res,True))
       (return (anSwarm,False))
    od"


(* 
  Design Definition of Function checkActorPluginFinished
*)
definition Des_checkActorPluginFinished :: "ACB ⇒ ((scheduler × RosLog), bool)nondet_monad"
  where "Des_checkActorPluginFinished anItr ≡
   do
    resi ← (whileLoopE (λi σ. i <  length (plugins (ACB.taskInfo anItr)))
             (λi. do
               pi ← return ((plugins (ACB.taskInfo anItr))!i);
               busName ← return (PluginInfo.busName pi);
               pluginName ← return (PluginInfo.name pi);
               resj ← (whileLoopE (λj σ. j < length (registeredBus (fst σ)))
                        (λj. do
                           bj ← gets (λσ. (registeredBus (fst σ))!j);
                           tempStr ← return (getBusName bj);
                           if tempStr = busName 
                           then 
                            (do
                              ret ← NonDetMonad_call.call (Des_getActorPluginState pluginName (ACB.actorID anItr))
                                     (λσ. ((registeredBus (fst σ))!j,snd σ)) (λσ b. σ);                     
                              if ret = pluginState.ACTOR_STATE_FINISH
                              then throwError j
                              else returnOk (j+1)
                             od)
                           else returnOk (j+1)
                         od)  (0) ) <catch> (λe. return e);
                condition (λσ. length (registeredBus (fst σ)) = resj )
                  (returnOk (i+1)) 
                  (throwError i)                  
              od) (0) ) <catch> (λe. return e);
    condition (λσ. length (plugins (ACB.taskInfo anItr)) = resi )
      (return False)
      (return True)
   od"
