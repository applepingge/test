theory SafetyModel
  imports Main
begin

locale safety = 
  fixes ReachableState  :: "'taskbinder × 'node ⇒ bool"

  ―‹1 taskbinder›
  fixes pActorScheduler :: "'taskbinder ⇒ 'sched"

  ―‹2 scheduler›
  fixes acbList :: "'sched ⇒ 'acb list"
  fixes swarmList :: "'sched ⇒ 'swarmInfo list"
  fixes actorNo::"'sched ⇒ int"
  fixes nSwarm::"'sched ⇒ int"
  fixes scheNum::"'sched ⇒ int"
  fixes registeredBus::"'sched ⇒ 'bus list"
  fixes scheMsgQueue::"'sched ⇒ 'schedmsg list"
  fixes toPauseActors::"'sched ⇒ 'actorPluginInfo list"
  fixes minPrioActors::"'sched ⇒ 'actorPluginInfo list"
  fixes finishingActors::"'sched ⇒ 'actorPluginInfo list"

  ―‹3 ACB›
  fixes ACBactorID::"'acb ⇒ int"
  fixes ACBname ::"'acb ⇒ string"
  fixes ACBtaskID::"'acb ⇒ int"
  fixes ACBtaskName::"'acb ⇒ string"
  fixes ACBtaskInfo::"'acb ⇒ 'taskInfo"
  fixes ACBstate ::"'acb ⇒ 'actorState"
  (*resources and configurable parameters*)
  (*currently not used*)
  fixes ACBsensors :: "'acb ⇒ 'SensorActuatorInfo list"
  fixes ACBactuators :: "'acb ⇒ 'SensorActuatorInfo list"
  (*platform information*)
  fixes ACBpPlatformInfo :: "'acb ⇒ 'PlatformInfo"
  fixes ACBpSwarmInfo :: "'acb ⇒ 'swarmInfo"
  (*priority*)
  fixes ACBprio :: "'acb ⇒ int"
  (*formation position*)
  fixes ACBformationPos :: "'acb ⇒ int" 
  (*formation type*)
  fixes ACBformationType :: "'acb ⇒ string"

  ―‹3 SwarmInfo›
  fixes swarmName ::"'swarmInfo ⇒ string"
  fixes swarmID::"'swarmInfo ⇒ int"
  fixes swarmPrio::"'swarmInfo ⇒ int"
  fixes swarmstate::"'swarmInfo ⇒ 'actorState"
  fixes generalActors::"'swarmInfo ⇒ string list"
  fixes excluActors::"'swarmInfo ⇒ string ⇒ string list option"
  fixes platformActors::"'swarmInfo ⇒ string ⇒ string list option"

  ―‹3 Bus›
  fixes busName ::"'bus ⇒ string"
  fixes busnRegisteredPlugin ::"'bus ⇒ int"
  fixes busPluginList ::"'bus ⇒ 'plugin list"
  fixes busPluginOwner ::"'bus ⇒ int list"
  fixes busPluginName ::"'bus ⇒ string list"

  ―‹4 TaskInfo›
  fixes taskInfoID::"'taskInfo ⇒ int"
  fixes actorName ::"'taskInfo ⇒ string"
  fixes taskXMLStr::"'taskInfo ⇒ string"
  fixes plugins :: "'taskInfo ⇒ 'pluginInfo list"

  ―‹4 Plugin›
  fixes pluginName :: "'plugin ⇒ string"
  fixes pluginActorID :: "'plugin ⇒ int"
  fixes pluginState :: "'plugin ⇒ 'pluginState"
  fixes pluginDuration :: "'plugin ⇒ int"

  ―‹5 PluginInfo›
  fixes pluginInfoName ::"'pluginInfo ⇒ string"
  fixes pluginInfoBusName ::"'pluginInfo ⇒ string"

  ―‹6 PluginState›
  fixes ACTOR_STATE_FINISH :: 'pluginState

  fixes ACTOR_STATE_FINISH' :: 'actorState
  fixes ACTOR_STATE_RUNNING' :: 'actorState
  fixes ACTOR_STATE_INIT' :: 'actorState

  ―‹ActorPluginInfo›
  fixes actorPluginInfoState :: "'actorPluginInfo ⇒ 'actorState"


assumes 
invariant_1:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
  int (length (acbList sche)) ≤ actorNo sche" and

invariant_2:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
  int (length (swarmList sche)) = nSwarm sche" and

invariant_3:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
  (∀acb ∈ set (acbList sche). 
            swarmName (ACBpSwarmInfo acb) ∈ set (map (λs. swarmName s) (swarmList sche)))" and

invariant_4:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (acbList sche). 
            swarmID (ACBpSwarmInfo acb) ∈ set (map (λs. swarmID s) (swarmList sche)))" and

invariant_5:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (acbList sche). 
      (∀plu ∈ set (plugins (ACBtaskInfo acb)). 
           pluginInfoBusName plu ∈ set (map (λb. busName b) (registeredBus sche))))" and

invariant_6:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (acbList sche).
               ACBactorID acb ≤ actorNo sche)" and

invariant_7:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀ swarm ∈ set (swarmList sche).
              swarmID swarm ≤ nSwarm sche)" and

invariant_8:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
distinct (map (λa. ACBactorID a) (acbList sche))" and

invariant_9:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
distinct (map (λa. ACBname a) (acbList sche))" and

invariant_10:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
distinct (map (λb. busName b) (registeredBus sche))" and

(*
invariant_11:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
distinct (map (λs. swarmName s) (swarmList sche))" and
*)

invariant_12:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
distinct (map (λs. swarmID s) (swarmList sche))" and

invariant_13:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (acbList sche). 
            ACBtaskID acb = taskInfoID (ACBtaskInfo acb))" and

invariant_14:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (acbList sche). 
            ACBname acb = actorName (ACBtaskInfo acb))" and

invariant_15:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀bus ∈ set (registeredBus sche). 
         ∀plu ∈ set (busPluginList bus).
                 (pluginState plu ≠ ACTOR_STATE_FINISH ⟶ 
                  pluginActorID plu ∈ set (map (λa. ACBactorID a) (acbList sche))))" and

invariant_16:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (acbList sche).                            
              distinct (map (λp. pluginInfoName p) (plugins (ACBtaskInfo acb))))" and

invariant_17:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (acbList sche). ACBprio acb ≤ MAX_PRIORITY)" and

invariant_18:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀swarm ∈ set (swarmList sche). swarmPrio swarm ≤ MAX_PRIORITY)" and

invariant_19:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (acbList sche). ACBstate acb ≠ ACTOR_STATE_FINISH')" and

invariant_20:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (finishingActors sche). actorPluginInfoState acb ≠ ACTOR_STATE_FINISH')" and

invariant_21:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (toPauseActors sche). actorPluginInfoState acb = ACTOR_STATE_RUNNING')" and

invariant_22:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (minPrioActors sche). actorPluginInfoState acb ≠ ACTOR_STATE_RUNNING')" and

invariant_23:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀bus ∈ set(registeredBus sche). (∀plu ∈ set (pluginList bus). pluginActorID plu ≤ actorNo sche))" and

invariant_24:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀acb ∈ set (acbList sche). (∀bus ∈ set(registeredBus sche) . ACBstate acb = ACTOR_STATE_INIT' ⟶ 
ACBactorID acb ∉ set (map pluginActorID (busPluginList bus))))" and

invariant_25:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
nSwarm sche ≤ SWARM_MAX_SIZE" and

invariant_26:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀bus∈ set(registeredBus sche). 
              busnRegisteredPlugin bus = int (length (busPluginList bus)))" and

invariant_27:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀bus∈ set(registeredBus sche). 
              nRegisteredPlugin bus = int (length (busPluginOwner bus)))" and

invariant_28:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀bus∈ set(registeredBus sche). 
              busnRegisteredPlugin bus = int (length (busPluginName bus)))" and

invariant_29:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀bus ∈ set (registeredBus sche). (∀i < length (busPluginName bus). 
    pluginActorID ((busPluginList bus)!i) = (busPluginOwner bus)!i))" and

invariant_30:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀bus ∈ set (registeredBus sche). (∀i < length (busPluginName bus). 
    pluginName ((busPluginList bus)!i) = (busPluginName bus)!i))" and

invariant_31:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀bus ∈ set (registeredBus sche). 
    distinct (map (λp. (pluginActorID p, pluginName p)) (busPluginList bus)))" and

invariant_32:"∀s. ReachableState s ⟶ sche = pActorScheduler (fst s) ⟶ 
(∀bus ∈ set (registeredBus sche). busnRegisteredPlugin bus < 128)"

begin
end
end
