lemma liftE_bind : 
  "doE  liftE A; B odE = do A ;B od  "
  using liftE_bindE by simp

lemma whenE_bind :
  "doE whenE A (doE liftE B;throwError e odE); C odE
   = (if A then (do B;throwError e od)else C) "
   apply(case_tac "A") apply auto
  unfolding whenE_def liftE_def bindE_def throwError_def returnOk_def
  by (metis (mono_tags, lifting) NonDetMonad.bind_assoc arg_cong_bind1 lift_Inl lift_Inr return_bind throwError_bind throwError_def)

lemma unlessE_bind :
  "doE unlessE A (throwError e); C odE
   = (if A then C else (throwError e )) "
 apply(case_tac "A") by auto

lemma condtion_sikpE_throwError:
   " doE y <- condition A (throwError a) skipE;throwError b odE
    = (condition A (throwError a) (throwError b)) " 
  apply(unfold condition_def throwError_def skipE_def bindE_def NonDetMonad.bind_def NonDetMonad.lift_def returnOk_def return_def)
  by auto

lemma liftE_catch_return : 
  "(doE m;
        liftE f
    odE <catch> return) =
   (do
      x ← m ;
      case x of Inl e ⇒ return e
              | Inr v' ⇒ f 
    od)" 
  unfolding NonDetMonad.bind_def  NonDetMonad.bindE_def catch_def K_bind_def
            NonDetMonad.lift_def liftE_def return_def throwError_def case_prod_unfold
  by (force split: sum.splits)

lemma liftE_return_returnOk : 
  "liftE (do guard A;return a od) = do guard A;returnOk a od"
  by (metis bind_K_bind condition_fail_rhs liftE_condition_distrib liftE_fail liftE_foo_to_fooE(3))
