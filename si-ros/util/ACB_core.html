theory ACB_core
imports "safety/SafetyModel"
begin

(* int type *)
type_synonym int64_t = int
type_synonym int32_t = int
type_synonym int16_t = int

(* Vector implementation under Isabelle *)
type_synonym 's vector = "'s list"
definition push_vec :: "'s vector ⇒ 's ⇒ 's vector" where
  "push_vec vec x ≡ rev (x # (rev vec))"
definition pop_back :: "'s vector ⇒ 's vector" where
  "pop_back vec ≡ rev (tl (rev vec))"
definition clear :: "'s vector ⇒ 's vector" where
  "clear vec ≡ []"
definition vector_empty :: "'s vector ⇒ bool" where
  "vector_empty vec ≡ length vec = 0"
definition size :: "'s vector ⇒ nat" where
  "size vec ≡ length vec"

(* Queue implementation under Isabelle *)
type_synonym 's queue = "'s list"
definition push_que :: "'s queue ⇒ 's ⇒ 's queue" 
  where "push_que que x ≡ que @ [x]"
definition front :: "'s queue ⇒ 's " 
  where "front que ≡ hd que"
definition pop :: "'s queue ⇒ 's queue" 
  where "pop que ≡ tl que"
definition queue_empty :: "'s queue ⇒ bool" where
  "queue_empty que ≡ length que = 0"


(* Actor Constant *)
definition MAX_PRIORITY :: "int" where
  "MAX_PRIORITY ≡ 32768"
definition ACTOR_STATE_INIT :: "int" where
  "ACTOR_STATE_INIT ≡ 0"
definition ACTOR_STATE_RUNNING :: "int" where
  "ACTOR_STATE_RUNNING ≡ 1"
definition ACTOR_STATE_PAUSE :: "int" where
  "ACTOR_STATE_PAUSE ≡ 2"
(*
definition ACTOR_STATE_READY :: "int" where
  "ACTOR_STATE_READY ≡ 3"
*)
definition ACTOR_STATE_FINISH :: "int" where
  "ACTOR_STATE_FINISH ≡ 4"
(* actor State *)
datatype actorState =  ACTOR_STATE_INIT | ACTOR_STATE_RUNNING |  
                       ACTOR_STATE_PAUSE | ACTOR_STATE_FINISH | 
                       NOT_VALID
(* Plugin State *)
datatype pluginState = ACTOR_STATE_RUNNING |  ACTOR_STATE_PAUSE | 
                        ACTOR_STATE_FINISH | NOT_VALID 


(* Actor Types *)
record SensorActuatorInfo =
  id :: int64_t
  type :: string
  properties :: "string ⇒ string option"
  params :: "string ⇒ string option"

record ResMatchResult =
  keyToRes :: "string ⇒ SensorActuatorInfo option"

record PluginInfo =
  name :: string
  busName :: string
definition PluginInfo :: "string ⇒ string ⇒ PluginInfo" where
  "PluginInfo aName aBusName = ⦇name=aName, busName=aBusName⦈"

record TaskInfo =
  taskID :: int64_t
  actorName :: string
  taskXMLStr :: string
  plugins :: "PluginInfo list"

record SwarmInfo =
  swarmName :: string
  swarmID :: int64_t
  swarmPrio :: int32_t
  state :: actorState
  generalActors :: "string vector"
  excluActors :: "string ⇒ string vector option"
  platformActors :: "string ⇒ string vector option"

definition initSwarmInfo :: "SwarmInfo" 
  where "initSwarmInfo ≡ 
          ⦇SwarmInfo.swarmName = [],
                     swarmID = -1,
                     swarmPrio = -1,
                     state = actorState.ACTOR_STATE_INIT,
                     generalActors = [],
                     excluActors = Map.empty ,
                     platformActors = Map.empty⦈" 

record PlatformInfo =
  id :: int64_t
  name :: string
  type :: string
  properties :: "string ⇒ string option"
definition initPlatformInfo :: " PlatformInfo" 
  where "initPlatformInfo ≡ 
       ⦇PlatformInfo.id = -1,
                     name = [],
                     type = [],
                     properties = Map.empty⦈"


record ACB =
  (*actorID and name*)
  actorID :: int64_t
  name :: string
  (*task info*)
  taskID :: int64_t
  taskName :: string
  taskInfo :: TaskInfo
  (*
  (*ACB tree info*)
  (*currently not used,a vecor is used*)
  pParent :: ACB option
  pChild :: ACB
  pSibling :: ACB
  *)
  (*actor state*)
  state :: actorState
  (*resources and configurable parameters*)
  (*currently not used*)
  sensors :: "SensorActuatorInfo vector"
  actuators :: "SensorActuatorInfo vector"
  (*platform information*)
  pPlatformInfo :: "PlatformInfo"
  pSwarmInfo :: "SwarmInfo"
  (*priority*)
  prio :: int32_t
  (*formation position*)
  formationPos :: int32_t 
  (*formation type*)
  formationType :: string

record ActorPluginInfo =
  actorID :: int64_t
  actorName :: string
  state :: actorState
  taskInfo :: TaskInfo


(* Scheduler Constant *)
definition SCHEDULE_CMD_RAW :: "int32_t"
  where "SCHEDULE_CMD_RAW ≡ 1"
definition SCHEDULE_CMD_SWITCH_PRIO :: "int32_t"
  where "SCHEDULE_CMD_SWITCH_PRIO ≡ 2"
definition SCHEDULE_CMD_ACTIVATE_ACTOR :: "int32_t"
  where "SCHEDULE_CMD_ACTIVATE_ACTOR ≡ 3"
definition SCHEDULE_CMD_YIELD_ACTOR :: "int32_t"
  where "SCHEDULE_CMD_YIELD_ACTOR ≡ 4"

datatype ScheduleMsgType = SCHEDULE_CMD_RAW| SCHEDULE_CMD_SWITCH_PRIO| 
                           SCHEDULE_CMD_ACTIVATE_ACTOR| SCHEDULE_CMD_YIELD_ACTOR

(* Scheduler Msg *)
record ScheduleMsg =
  type :: ScheduleMsgType
  param1 :: int64_t
  param2 :: int64_t
  param3 :: string

definition nullScheduleMsg :: "ScheduleMsg" where
  "nullScheduleMsg ≡ ⦇ type = SCHEDULE_CMD_RAW,
                       param1 = -1,
                       param2 = -1,
                       param3 = []
                      ⦈"

(*task bind srv*)
datatype task_bind = string | bool
