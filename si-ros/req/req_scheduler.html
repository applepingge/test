(* 
  Requirement Definition of findFinishActors
*)    
definition Re_findFinishActors :: "scheduler ⇒ scheduler"
  where "Re_findFinishActors sche ≡
              let 
                  al' = filter (λa. ACB.state a ≠ actorState.ACTOR_STATE_INIT ∧ Re_checkActorPluginFinished sche a) (acbList sche);
                  al'' = map (λa. ⦇ActorPluginInfo.actorID   = ACB.actorID a,
                                  ActorPluginInfo.actorName = ACB.name a,
                                  ActorPluginInfo.state     = ACB.state a,
                                  ActorPluginInfo.taskInfo  = ACB.taskInfo a⦈) 
                              al'                  
              in sche⦇finishingActors := al''⦈"
definition findFinishActors_invariant :: "scheduler ⇒ bool"
  where "findFinishActors_invariant sche ≡ 
               (∀a'∈set (finishingActors sche). 
                   ∃a ∈ set (acbList sche). 
                   ActorPluginInfo.actorName a' = ACB.name a
                   ∧ ActorPluginInfo.actorID a' = ACB.actorID a
               )"
lemma "Re_findFinishActors sche = sche' ⟹  acbList sche = acbList sche' ∧ findFinishActors_invariant sche'"
  apply (simp add: Re_findFinishActors_def findFinishActors_invariant_def)
  by fastforce
  

(* 
  Requirement Definition of delFinishActors
*)    
definition Re_delFinishActors :: "scheduler ⇒ scheduler" 
  where "Re_delFinishActors sche ≡ 
                 let (sche', flag) = Re_sche_updateActorPluginState sche (finishingActors sche) actorState.ACTOR_STATE_FINISH
                 in (if flag
                     then sche'⦇acbList := filter (λa. ACB.actorID a ∉ set (map (λapi. ActorPluginInfo.actorID api) (finishingActors sche'))) (acbList sche')⦈
                     else sche'
                 )"


(* 
  Requirement Definition of findMinPrioSwarm
*)    
definition Re_findMinPrioSwarm :: "SwarmInfo list  ⇒ int ⇒ (SwarmInfo option × int)" 
  where "Re_findMinPrioSwarm sil minPrio ≡ 
                let minPrioSwarm = find (λsi. (swarmPrio si) < minPrio ∧ (∀si'∈ set sil. (swarmPrio si) ≤ (swarmPrio si'))) sil
                in
                   (minPrioSwarm, if minPrioSwarm = None then minPrio else swarmPrio (the minPrioSwarm))"


(* 
  Requirement Definition of ifMinPrioSwarmHaveActor
*)    
definition ifMinPrioSwarmHaveActor :: "ACB list ⇒ SwarmInfo ⇒ int ⇒ int ⇒ (int × int)"
  where "ifMinPrioSwarmHaveActor al minSwarmPtr minPrio minIndex ≡ 
                let minPrioACB = find (λa. swarmID (pSwarmInfo a) = swarmID minSwarmPtr ∧ 
                                        (ACB.prio a) < minPrio ∧ 
                                            (∀j<length al. swarmID (pSwarmInfo (al!j)) = swarmID minSwarmPtr ⟶
                                                   (ACB.prio a) ≤ (ACB.prio (al!j)))) al
                in
                   (if minPrioACB ≠ None
                    then (int (index al (the minPrioACB)),ACB.prio (the minPrioACB))
                    else (minIndex,minPrio)
                    )"

definition Re_schedule_1 :: "scheduler ⇒ (scheduler×bool)"
  where "Re_schedule_1 sche ≡ 
             let sche1 = sche⦇scheNum := scheNum sche + 1⦈;
                 condition1 = (length (acbList sche1) = 0)
             in (sche1,condition1)"
definition Re_schedule_2 :: "scheduler ⇒ scheduler"
  where "Re_schedule_2 sche ≡ 
             let
                sche' = sche⦇finishingActors := []⦈;
                sche'' = Re_findFinishActors sche';
                condition2 = (length (finishingActors sche'') ≠ 0);
                sche''' = (if condition2 then Re_delFinishActors sche'' else sche'') 
             in sche'''"
definition Re_schedule_3 :: "scheduler ⇒ (SwarmInfo option × int × int × bool × bool)"
  where "Re_schedule_3 sche3 ≡ 
             let
                 (minSwarmPtr, minSwarmPrio) = Re_findMinPrioSwarm (swarmList sche3) (MAX_PRIORITY+1);
                 condition3 = (minSwarmPtr = None);
                 (minIdx, minPrio) = (ifMinPrioSwarmHaveActor (acbList sche3) (the minSwarmPtr) (MAX_PRIORITY+1) (-1));
                 condition4 = (minIdx = -(1))
             in (minSwarmPtr, minSwarmPrio,  minPrio, condition3, condition4)"
definition Re_schedule_4 :: "scheduler ⇒ bool ⇒ SwarmInfo option ⇒ SwarmInfo option ⇒ scheduler"
  where "Re_schedule_4 sche5 condition6 minSwarmPtr runningSwarmPtr ≡ 
             let (sche5', flag) = Re_sche_updateActorPluginState sche5 (toPauseActors sche5) (actorState.ACTOR_STATE_PAUSE);
                 sche6 = (if condition6
                          then 
                             if flag
                             then Re_pauseSwarmAndActors sche5' runningSwarmPtr (the minSwarmPtr)
                             else sche5'
                          else sche5 )
             in sche6"
definition Re_schedule_5 :: "scheduler ⇒ bool ⇒ SwarmInfo option ⇒ scheduler"
  where "Re_schedule_5 sche6 condition7 minSwarmPtr ≡ 
             let (sche6', flag) = Re_sche_updateActorPluginState sche6 (minPrioActors sche6) (actorState.ACTOR_STATE_RUNNING);
                 sche7 = (if condition7
                          then 
                             if flag
                             then Re_runSwarmAndActors sche6' (the minSwarmPtr)
                             else sche6'
                          else sche6 )
             in sche7"

definition Re_schedule :: "scheduler ⇒ scheduler"
  where "Re_schedule sche ≡ 
            let (sche1,condition1) = Re_schedule_1 sche;
                sche3 = Re_schedule_2 sche1;
                (minSwarmPtr, minSwarmPrio, minPrio, condition3, condition4) = Re_schedule_3 sche3;
                sche3' = sche3⦇minPrioActors := []⦈;
                sche4 = Re_findAcotorUnderMinSwarm sche3' (the minSwarmPtr) minPrio;
                (runningSwarmPtr, nRunningSwarm) = Re_findRunningSwarm (swarmList sche4) 0;
                condition5 = (nRunningSwarm > 1);
                sche4' = sche4⦇toPauseActors := []⦈;
                sche5 = Re_findToPauseActors sche4' (the minSwarmPtr) minPrio;
                condition6 = (length (toPauseActors sche5) ≠ 0);
                sche6 = Re_schedule_4 sche5 condition6 minSwarmPtr runningSwarmPtr;
                condition7 = (length (minPrioActors sche6) ≠ 0);
                sche7 = Re_schedule_5 sche6 condition7 minSwarmPtr
            in 
              (if condition1
              then sche1
              else (
                if condition3 ∨ condition4
                then sche3
                else (
                  if condition5
                  then sche4
                  else sche7)))"
