theory req_bus

imports "req_plugin"

begin


(* 
  Requirement Definition of Function createInstance
*)
definition Re_createInstance :: "bus ⇒ string ⇒ (bus × bool)"
  where "Re_createInstance b pluName ≡ 
             let plug = nullPlugin⦇plugin.name := pluName⦈
             in
              (if pluName = []
               then (b, False)
               else (b⦇pluginList := (pluginList b) @ [plug]⦈, True)
              )"


(* 
  Requirement Definition of Function loadPlugin 
*)
definition Re_loadPlugin :: "bus ⇒ string ⇒ int64_t ⇒ (bus × bool)"
  where "Re_loadPlugin b pName actID ≡
             let
                (b', flag) = Re_createInstance b pName;
                b'' = b'⦇ pluginOwner := (pluginOwner b) @ [actID],
                         pluginName := (pluginName b) @ [pName]⦈ ;
                list = pluginList b'' ;
                num = length (pluginList b'') - 1 ;
                b''' = b''⦇pluginList := list[num := pluInitialize (list!num) actID pName]⦈ ;
                b'''' = b'''⦇nRegisteredPlugin := nRegisteredPlugin b''' +1⦈ ;
                res = (if ¬flag
                       then (b', False)
                       else (b'''', True)
                      )
             in
                (if nRegisteredPlugin b < PLUGIN_NUM_MAX - 1
                 then res
                 else (b, False)
                )"


(* 
  Requirement Definition of Function startPlugin
*)
definition Re_startPlugin :: "bus ⇒ string ⇒ int64_t ⇒ (bus × bool)"
  where "Re_startPlugin b pName actID ≡
           let 
              num = find_index (λp. plugin.actorID p = actID ∧ plugin.name p = pName) (pluginList b)
           in 
             ( if (int)num = nRegisteredPlugin b 
               then (b, False)
               else (b⦇pluginList := (pluginList b)[num := Re_resume ((pluginList b)!num)] ⦈, True)
             )" 


(* 
  Requirement Definition of Function pausePlugin
*)
definition Re_pausePlugin :: "bus ⇒ string ⇒ int64_t ⇒ (bus × bool)"
  where "Re_pausePlugin b pName actID ≡
           let 
              num = find_index (λp. plugin.actorID p = actID ∧ plugin.name p = pName) (pluginList b)
           in 
              ( if (int)num = nRegisteredPlugin b 
                then (b, False)
                else (b⦇pluginList := (pluginList b)[num := Re_pause ((pluginList b)!num)] ⦈, True)
              )"


(* 
  Requirement Definition of Function stopPlugin
*)
definition Re_stopPlugin :: "bus ⇒ string ⇒ int64_t ⇒ (bus × bool)"
  where "Re_stopPlugin b pName actID ≡
           let 
              num = find_index (λp. plugin.actorID p = actID ∧ plugin.name p = pName) (pluginList b)
           in 
              ( if (int)num = nRegisteredPlugin b 
                then (b, False)
                else (b⦇pluginList := (pluginList b)[num := Re_stop ((pluginList b)!num)] ⦈, True)
              )"


(* 
  Requirement Definition of Function startActorPlugins
*)
definition Re_startActorPlugins :: "bus ⇒ int64_t ⇒ (bus × bool)"
    where "Re_startActorPlugins b actID ≡
                (b⦇pluginList := map (λp. if actorID p = actID then Re_resume p else p) (pluginList b)⦈, True)"


(* 
  Requirement Definition of Function pauseActorPlugins
*)
definition Re_pauseActorPlugins :: "bus ⇒ int64_t ⇒ (bus × bool)"
    where "Re_pauseActorPlugins b actID ≡
                (b⦇pluginList := map (λp. if actorID p = actID then Re_pause p else p) (pluginList b)⦈, True)"


(* 
  Requirement Definition of Function stopActorPlugins
*)
definition Re_stopActorPlugins :: "bus ⇒ int64_t ⇒ (bus × bool)"
    where "Re_stopActorPlugins b actID ≡
                (b⦇pluginList := map (λp. if actorID p = actID then Re_stop p else p) (pluginList b)⦈, True)"


(* 
  Requirement Definition of Function updateActorPluginState
*)
fun loadPluginInfoList :: "PluginInfo list ⇒ bus ⇒ int64_t ⇒ (bus × bool)"
  where "loadPluginInfoList [] b actID = (b, True)"|
        "loadPluginInfoList (p#pl) b actID = 
         (if busName p = bus.name b
          then 
              (let (b', flag) = Re_loadPlugin b (PluginInfo.name p) actID
               in
                  (if ¬flag
                   then (b, False)
                   else loadPluginInfoList pl b' actID
                  )
              )
          else loadPluginInfoList pl b actID)"
definition Re_updateActorPluginState :: "bus ⇒ ActorPluginInfo ⇒ actorState ⇒ (bus × bool)"
  where "Re_updateActorPluginState b act st ≡
             (case st of 
              actorState.ACTOR_STATE_PAUSE ⇒ (Re_pauseActorPlugins b (ActorPluginInfo.actorID act)) |
              actorState.ACTOR_STATE_FINISH ⇒ Re_stopActorPlugins b (ActorPluginInfo.actorID act)|
              actorState.ACTOR_STATE_INIT ⇒ 
                   loadPluginInfoList (plugins (ActorPluginInfo.taskInfo act)) b (ActorPluginInfo.actorID act) |
              actorState.ACTOR_STATE_RUNNING ⇒
                   (let 
                      (b', flag) = 
                        (if ActorPluginInfo.state act = actorState.ACTOR_STATE_INIT 
                        then loadPluginInfoList (plugins (ActorPluginInfo.taskInfo act))
                                                b (ActorPluginInfo.actorID act)
                        else (b, True))
                    in 
                       if flag
                       then Re_startActorPlugins b' (ActorPluginInfo.actorID act)
                       else (b', False)
                    ) |
              _ ⇒ (b, False))"


(* 
  Requirement Definition of Function updateActorPluginStates
*)
fun Re_updateActorPluginStates :: "bus ⇒ ActorPluginInfo list ⇒ actorState ⇒ (bus × bool)"
  where "Re_updateActorPluginStates b [] sta = (b, True)" |
        "Re_updateActorPluginStates b (a#al) sta = 
           (if ActorPluginInfo.state a = sta
            then Re_updateActorPluginStates b al sta
            else 
              (if ¬ (snd (Re_updateActorPluginState b a sta))
               then Re_updateActorPluginState b a sta
               else Re_updateActorPluginStates (fst (Re_updateActorPluginState b a sta)) al sta
              )
           )"


(* 
  Requirement Definition of Function getActorPlugin
*)
definition Re_getActorPlugin :: "bus ⇒ string ⇒ int64_t ⇒ (plugin × bool)"
  where "Re_getActorPlugin b plugName actID ≡
          let 
             num = find_index (λp. plugin.name p = plugName ∧ plugin.actorID p = actID) (pluginList b)
          in 
             ( if (int)num = nRegisteredPlugin b
               then (nullPlugin, False)
               else ((pluginList b)!num, True)
             )"


(* 
  Requirement Definition of Function getActorPluginState
*)
definition Re_getActorPluginState :: "bus ⇒ string ⇒ int64_t ⇒ pluginState"
  where "Re_getActorPluginState b plugName actID ≡
          let 
             num = find_index (λp. plugin.name p = plugName ∧ plugin.actorID p = actID) (pluginList b)
          in 
             ( if (int)num = nRegisteredPlugin b
               then pluginState.NOT_VALID
               else getPlugState ((pluginList b)!num)
             )"
