lemma corres_condition:
  "⟦ ∀(s,t)∈sr. G s = G' t;
     corres_underlying sr nf nf' r P P' a c; 
     corres_underlying sr nf nf' r P P' b d ⟧ ⟹
    corres_underlying sr nf nf' r
      P  P'
        (condition G a b) (condition G' c d)"
  unfolding condition_def corres_underlying_def 
  apply simp 
  unfolding case_prod_unfold
  by auto

  
lemma des_imp_initializePlugin:
  " int64Equal actID actIDC ∧ stringEqual pName pNameC ⟹ 
    ⦃λs. pluginEqual pluC s plu⦄
        initializePlugin' pluC actIDC pNameC
    ⦃λr s. pluginEqual pluC s (pluInitialize plu actID pName)⦄"
  apply(unfold initializePlugin'_def pluInitialize_def)
  apply wpsimp
  apply (unfold pluginEqual_def int64Equal_def statePluginEqual_def int32Equal_def)
  apply simp_all
  proof -
  fix s :: lifted_globals
  assume a1: "stringEqual pName pNameC"
  obtain ww :: "GeneralPlugin_C ⇒ 32 signed word" and wwa :: "GeneralPlugin_C ⇒ 32 signed word" and wwb :: "GeneralPlugin_C ⇒ 64 signed word" and pp :: "GeneralPlugin_C ⇒ 8 word ptr" where
  "∀g. g = GeneralPlugin_C (ww g) (wwa g) (wwb g) (pp g)"
    by (meson GeneralPlugin_C.exhaust)
    then have f2: "∀g. GeneralPlugin_C (duration_C g) (GeneralPlugin_C.state_C g) (GeneralPlugin_C.actorID_C g) (GeneralPlugin_C.name_C g) = g"
      by (metis GeneralPlugin_C.actorID_C.simps GeneralPlugin_C.name_C.simps GeneralPlugin_C.state_C.simps duration_C.simps)
    then have f3: "∀g f. GeneralPlugin_C.state_C (duration_C_update f g) = GeneralPlugin_C.state_C g"
      by (metis (no_types) GeneralPlugin_C.state_C.simps duration_C_update.simps)
    have f4: "∀g f. GeneralPlugin_C.actorID_C (duration_C_update f g) = GeneralPlugin_C.actorID_C g"
      using f2 by (metis GeneralPlugin_C.actorID_C.simps duration_C_update.simps)
    have f5: "∀g f. GeneralPlugin_C.actorID_C (GeneralPlugin_C.state_C_update f g) = GeneralPlugin_C.actorID_C g"
      using f2 by (metis GeneralPlugin_C.actorID_C.simps GeneralPlugin_C.state_C_update.simps)
    have f6: "∀g f. GeneralPlugin_C.actorID_C (GeneralPlugin_C.name_C_update f g) = GeneralPlugin_C.actorID_C g"
      using f2 by (metis (no_types) GeneralPlugin_C.actorID_C.simps GeneralPlugin_C.name_C_update.simps)
    have f7: "∀g f. GeneralPlugin_C.name_C (duration_C_update f g) = GeneralPlugin_C.name_C g"
      using f2 by (metis GeneralPlugin_C.name_C.simps duration_C_update.simps)
    have f8: "∀g f. GeneralPlugin_C.name_C (GeneralPlugin_C.state_C_update f g) = GeneralPlugin_C.name_C g"
      using f2 by (metis (no_types) GeneralPlugin_C.name_C.simps GeneralPlugin_C.state_C_update.simps)
    have f9: "∀g f. f (GeneralPlugin_C.name_C g) = GeneralPlugin_C.name_C (GeneralPlugin_C.name_C_update f g)"
      using f2 by (metis (no_types) GeneralPlugin_C.name_C.simps GeneralPlugin_C.name_C_update.simps)
    have f10: "∀g f. f (GeneralPlugin_C.actorID_C g) = GeneralPlugin_C.actorID_C (GeneralPlugin_C.actorID_C_update f g)"
  using f2 by (metis (no_types) GeneralPlugin_C.actorID_C.simps GeneralPlugin_C.actorID_C_update.simps)
  have f11: "∀g f. f (GeneralPlugin_C.state_C g) = GeneralPlugin_C.state_C (GeneralPlugin_C.state_C_update f g)"
    using f2 by (metis (no_types) GeneralPlugin_C.state_C.simps GeneralPlugin_C.state_C_update.simps)
    have "sint (duration_C (duration_C_update (λw. 1) (GeneralPlugin_C.state_C_update (λw. impl_ACTOR_STATE_PAUSE) (GeneralPlugin_C.name_C_update (λp. pNameC) (GeneralPlugin_C.actorID_C_update (λw. actIDC) (heap_GeneralPlugin_C s pluC)))))) = 1"
      using f2 by (metis (no_types) duration_C.simps duration_C_update.simps sint_1_eq_1_x(3))
      then show "actIDC = GeneralPlugin_C.actorID_C (duration_C_update (λw. 1) (GeneralPlugin_C.state_C_update (λw. impl_ACTOR_STATE_PAUSE) (GeneralPlugin_C.name_C_update (λp. pNameC) (GeneralPlugin_C.actorID_C_update (λw. actIDC) (heap_GeneralPlugin_C s pluC))))) ∧ stringEqual pName (GeneralPlugin_C.name_C (duration_C_update (λw. 1) (GeneralPlugin_C.state_C_update (λw. impl_ACTOR_STATE_PAUSE) (GeneralPlugin_C.name_C_update (λp. pNameC) (GeneralPlugin_C.actorID_C_update (λw. actIDC) (heap_GeneralPlugin_C s pluC)))))) ∧ sint (GeneralPlugin_C.state_C (duration_C_update (λw. 1) (GeneralPlugin_C.state_C_update (λw. impl_ACTOR_STATE_PAUSE) (GeneralPlugin_C.name_C_update (λp. pNameC) (GeneralPlugin_C.actorID_C_update (λw. actIDC) (heap_GeneralPlugin_C s pluC)))))) = 2 ∧ sint (duration_C (duration_C_update (λw. 1) (GeneralPlugin_C.state_C_update (λw. impl_ACTOR_STATE_PAUSE) (GeneralPlugin_C.name_C_update (λp. pNameC) (GeneralPlugin_C.actorID_C_update (λw. actIDC) (heap_GeneralPlugin_C s pluC)))))) = 1"
        using f11 f10 f9 f8 f7 f6 f5 f4 f3 a1 ded_imp_pluginPAUSE by presburger
  qed 
  
  
lemma des_imp_pause:
  "corres_underlying ({(t,s). pluginEqual plu s (fst t)}) False False (λ_ _. True) (λ_. True) (λ_. True)  (Des_pause) (pause' plu)"
  apply (unfold pause'_def Des_pause_def)
   apply(unfold pluginEqual_def statePluginEqual_def setState'_def)
   apply(unfold corres_underlying_def NonDetMonad.bind_def condition_def modify_def guard_def return_def skip_def gets_def get_def put_def setPlugState_def) 
  apply simp_all apply rule+
  apply (metis (no_types) Plustate_C_update2)
  apply (metis (no_types) ded_imp_pluginFINISH word_sint.Rep_inject)
  apply rule+
  apply (metis (no_types) Plustate_C_update2)
  apply (metis Plustate_C_update1 Plustate_C_update3 Plustate_C_update4 ded_imp_pluginPAUSE)
  apply rule+
  using ded_imp_pluginPAUSE apply presburger
  using ded_imp_pluginPAUSE apply blast
  apply rule+
  using ded_imp_pluginFINISH apply auto[1]
  by (metis (full_types) Plustate_C_update1 Plustate_C_update2 Plustate_C_update3 Plustate_C_update4 ded_imp_pluginPAUSE)

lemma des_imp_stop:
  "corres_underlying ({(s,t). pluginEqual plu t (fst s)}) False False (λ_ _. True) (λ_. True) (λ_. True) (Des_stop)  (stop' plu)"
  apply(unfold stop'_def Des_stop_def setState'_def)
  apply(rule corres_symb_exec_r[where Q' = "λr v. True"])
     apply(unfold K_bind_def)
     defer 
     apply wpsimp+
  apply (rule corres_condition)
  subgoal
    apply (unfold pluginEqual_def statePluginEqual_def) apply auto
    using ded_imp_pluginFINISH apply auto[1]
    using ded_imp_pluginFINISH apply auto[1]
    using ded_imp_pluginFINISH word_sint.Rep_inject apply fastforce
    using ded_imp_pluginFINISH by auto[1]
  subgoal
    apply (unfold corres_underlying_def gets_def get_def put_def return_def skip_def  NonDetMonad.bind_def modify_def)
    by auto
  apply(rule corres_symb_exec_r[where Q' = "λr v. True"])
     defer 
  apply wpsimp+
  apply(rule corres_symb_exec_l[where Q = "λr v. True"])
  subgoal for st
    apply (rule corres_modify')
    apply auto
    apply (unfold pluginEqual_def statePluginEqual_def setPlugState_def)
    apply auto
    by (metis Plustate_C_update2,metis Plustate_C_update3,
        metis Plustate_C_update1 ded_imp_pluginFINISH,metis Plustate_C_update4)+
  by wpsimp+
