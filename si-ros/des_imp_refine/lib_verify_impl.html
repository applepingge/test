definition nextAddrPointer :: "8 word ptr ⇒ 8 word ptr" 
  where "nextAddrPointer pointer ≡ pointer  +⇩p 1"
function nameEqualAux :: "string ⇒ 8 word ptr ⇒ nat ⇒ nat ⇒ bool"
  where "nameEqualAux str strPointer i strLen = (if i = strLen ∧ char_of_integer (integer_of_int (sint (ptr_val strPointer))) = char_of_integer 0
                                             then True
                                             else 
                                                (if i > strLen ∨ char_of_integer (integer_of_int (sint (ptr_val strPointer))) = char_of_integer 0
                                                 then False
                                                 else (
                                                      if (str!i) = char_of_integer (integer_of_int (sint (ptr_val strPointer)))
                                                      then nameEqualAux str (nextAddrPointer strPointer) (i+1) strLen
                                                      else False
                                                      )
                                                )
                                             )"
  using prod_cases4 apply blast
  by force

function stringAux :: "string ⇒ nat ⇒ int ⇒ 8 word ptr ⇒ lifted_globals ⇒ bool "
  where "stringAux str i len strPointer glo = 
          (if i = len 
           then True 
           else ( if (str!i) = char_of_integer (integer_of_int (sint (heap_w8 glo (strPointer +⇩p i))))  
                  then True ∧ (stringAux str (i+1) len strPointer glo)
                  else False) ) "
  using prod_cases4 
  by auto
 

definition stringEqual' :: "string ⇒ 8 word ptr ⇒ lifted_globals ⇒ bool"
  where "stringEqual' str strPointer glo ≡

          else False "
